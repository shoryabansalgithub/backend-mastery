## 2. Drizzle Schema (`src/db/schema.ts`)

```typescript
import {
  pgTable, uuid, text, timestamp, integer,
  boolean, jsonb, primaryKey, index
} from 'drizzle-orm/pg-core';
import { relations } from 'drizzle-orm';

// ============================================================
// SHARED COLUMNS (via helper)
// ============================================================
// Every tenant-scoped table gets: id, orgId, createdAt, updatedAt, deletedAt
const timestamps = {
  createdAt: timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
  updatedAt: timestamp('updated_at', { withTimezone: true }).notNull().defaultNow(),
  deletedAt: timestamp('deleted_at', { withTimezone: true }),  // soft delete
};

// ============================================================
// ORGANIZATIONS
// ============================================================
export const organizations = pgTable('organizations', {
  id:        uuid('id').primaryKey().defaultRandom(),
  name:      text('name').notNull(),
  slug:      text('slug').notNull().unique(),  // URL-safe identifier
  plan:      text('plan').notNull().default('free'),
  ...timestamps,
});

// ============================================================
// USERS
// ============================================================
export const users = pgTable('users', {
  id:           uuid('id').primaryKey().defaultRandom(),
  email:        text('email').notNull().unique(),
  passwordHash: text('password_hash').notNull(),
  name:         text('name').notNull(),
  ...timestamps,
});

// ============================================================
// ORG MEMBERS (Users ↔ Organizations junction)
// ============================================================
export const orgMembers = pgTable('org_members', {
  orgId:     uuid('org_id').notNull().references(() => organizations.id, { onDelete: 'cascade' }),
  userId:    uuid('user_id').notNull().references(() => users.id, { onDelete: 'cascade' }),
  role:      text('role').notNull().default('member'),  // owner | admin | member | viewer
  joinedAt:  timestamp('joined_at', { withTimezone: true }).notNull().defaultNow(),
}, (t) => ({
  pk: primaryKey({ columns: [t.orgId, t.userId] }),
  orgIdx: index('idx_org_members_org').on(t.orgId),
  userIdx: index('idx_org_members_user').on(t.userId),
}));

// ============================================================
// PROJECTS
// ============================================================
export const projects = pgTable('projects', {
  id:          uuid('id').primaryKey().defaultRandom(),
  orgId:       uuid('org_id').notNull().references(() => organizations.id, { onDelete: 'cascade' }),
  name:        text('name').notNull(),
  description: text('description'),
  status:      text('status').notNull().default('active'),  // active | archived
  ...timestamps,
}, (t) => ({
  orgIdx: index('idx_projects_org').on(t.orgId),
  orgStatusIdx: index('idx_projects_org_status').on(t.orgId, t.status),
}));

// ============================================================
// TASKS
// ============================================================
export const tasks = pgTable('tasks', {
  id:          uuid('id').primaryKey().defaultRandom(),
  orgId:       uuid('org_id').notNull().references(() => organizations.id, { onDelete: 'cascade' }),
  projectId:   uuid('project_id').notNull().references(() => projects.id, { onDelete: 'cascade' }),
  assigneeId:  uuid('assignee_id').references(() => users.id),
  title:       text('title').notNull(),
  description: text('description'),
  status:      text('status').notNull().default('todo'),  // todo | in_progress | done | cancelled
  priority:    text('priority').notNull().default('medium'),  // low | medium | high | critical
  dueDate:     timestamp('due_date', { withTimezone: true }),
  version:     integer('version').notNull().default(1),  // for optimistic locking
  ...timestamps,
}, (t) => ({
  orgProjectIdx: index('idx_tasks_org_project').on(t.orgId, t.projectId),
  // GIN index for full-text search — created in migration, not here
  // because Drizzle doesn't support GIN on expressions directly
}));

// ============================================================
// LABELS
// ============================================================
export const labels = pgTable('labels', {
  id:     uuid('id').primaryKey().defaultRandom(),
  orgId:  uuid('org_id').notNull().references(() => organizations.id, { onDelete: 'cascade' }),
  name:   text('name').notNull(),
  color:  text('color').notNull().default('#6B7280'),
  ...timestamps,
}, (t) => ({
  orgIdx: index('idx_labels_org').on(t.orgId),
}));

// ============================================================
// TASK LABELS (Tasks ↔ Labels many-to-many)
// ============================================================
export const taskLabels = pgTable('task_labels', {
  taskId:  uuid('task_id').notNull().references(() => tasks.id, { onDelete: 'cascade' }),
  labelId: uuid('label_id').notNull().references(() => labels.id, { onDelete: 'cascade' }),
}, (t) => ({
  pk: primaryKey({ columns: [t.taskId, t.labelId] }),
}));

// ============================================================
// COMMENTS
// ============================================================
export const comments = pgTable('comments', {
  id:      uuid('id').primaryKey().defaultRandom(),
  orgId:   uuid('org_id').notNull().references(() => organizations.id, { onDelete: 'cascade' }),
  taskId:  uuid('task_id').notNull().references(() => tasks.id, { onDelete: 'cascade' }),
  userId:  uuid('user_id').notNull().references(() => users.id),
  body:    text('body').notNull(),
  ...timestamps,
}, (t) => ({
  taskIdx: index('idx_comments_task').on(t.taskId),
}));

// ============================================================
// AUDIT LOG
// ============================================================
export const auditEvents = pgTable('audit_events', {
  id:           uuid('id').primaryKey().defaultRandom(),
  orgId:        uuid('org_id').notNull().references(() => organizations.id),
  userId:       uuid('user_id').references(() => users.id),
  action:       text('action').notNull(),  // created | updated | deleted
  resourceType: text('resource_type').notNull(),  // task | project | comment | member
  resourceId:   uuid('resource_id').notNull(),
  before:       jsonb('before'),   // snapshot before change (null for creates)
  after:        jsonb('after'),    // snapshot after change (null for deletes)
  createdAt:    timestamp('created_at', { withTimezone: true }).notNull().defaultNow(),
}, (t) => ({
  orgIdx:      index('idx_audit_org').on(t.orgId),
  resourceIdx: index('idx_audit_resource').on(t.resourceType, t.resourceId),
}));

// ============================================================
// RELATIONS
// ============================================================
export const tasksRelations = relations(tasks, ({ one, many }) => ({
  project:  one(projects,  { fields: [tasks.projectId],  references: [projects.id] }),
  assignee: one(users,     { fields: [tasks.assigneeId], references: [users.id] }),
  comments: many(comments),
  labels:   many(taskLabels),
}));

export const taskLabelsRelations = relations(taskLabels, ({ one }) => ({
  task:  one(tasks,  { fields: [taskLabels.taskId],  references: [tasks.id] }),
  label: one(labels, { fields: [taskLabels.labelId], references: [labels.id] }),
}));
```

## 3. Migration Strategy

### `drizzle.config.ts`
```typescript
import { defineConfig } from 'drizzle-kit';
export default defineConfig({
  schema: './src/db/schema.ts',
  out: './src/db/migrations',
  dialect: 'postgresql',
  dbCredentials: { url: process.env.DATABASE_URL! },
});
```

### Migration generation
```bash
npx drizzle-kit generate   # produces SQL migration files
npx drizzle-kit migrate    # applies pending migrations
```

### GIN index for search (must be in manual migration)
Since Drizzle doesn't support `CREATE INDEX ... USING GIN (to_tsvector(...))`, add this to a separate migration file:
```sql
-- src/db/migrations/0002-search-indexes.sql
CREATE EXTENSION IF NOT EXISTS pg_trgm;

-- GIN trigram index for ILIKE-style search on tasks
CREATE INDEX idx_tasks_title_trgm ON tasks USING GIN (title gin_trgm_ops);
CREATE INDEX idx_tasks_description_trgm ON tasks USING GIN (description gin_trgm_ops);
CREATE INDEX idx_comments_body_trgm ON comments USING GIN (body gin_trgm_ops);
```

### Zero-downtime migration pattern (for adding NOT NULL columns)
```
Step 1: Add column as nullable   →  migration file A
Step 2: Backfill existing rows   →  migration file B (UPDATE)
Step 3: Add NOT NULL constraint  →  migration file C (ALTER TABLE)
Never combine these into one migration.
```

## 4. Tenant Scope Middleware

```typescript
// src/middleware/tenantScope.ts
import { Request, Response, NextFunction } from 'express';
import { db } from '../db';
import { orgMembers } from '../db/schema';
import { and, eq, isNull } from 'drizzle-orm';

export interface TenantContext {
  orgId: string;
  userId: string;
  role: 'owner' | 'admin' | 'member' | 'viewer';
}

// Extend Express Request type
declare global {
  namespace Express {
    interface Request {
      tenant: TenantContext;
    }
  }
}

export function requireTenantAccess(minRole: TenantContext['role'] = 'viewer') {
  return async (req: Request, res: Response, next: NextFunction) => {
    const orgId = req.params.orgId;
    const userId = req.user!.id;  // set by authenticate middleware

    const [membership] = await db
      .select({ role: orgMembers.role })
      .from(orgMembers)
      .where(and(
        eq(orgMembers.orgId, orgId),
        eq(orgMembers.userId, userId),
      ))
      .limit(1);

    if (!membership) {
      return res.status(403).json({ error: 'Not a member of this organization' });
    }

    const ROLE_RANK = { viewer: 0, member: 1, admin: 2, owner: 3 };
    if (ROLE_RANK[membership.role as keyof typeof ROLE_RANK] < ROLE_RANK[minRole]) {
      return res.status(403).json({ error: 'Insufficient permissions' });
    }

    req.tenant = { orgId, userId, role: membership.role as TenantContext['role'] };
    next();
  };
}
```

### Usage in routes
```typescript
// All org-scoped routes use requireTenantAccess

  authenticate,
  requireTenantAccess('viewer'),
  listProjects
);
```

## 5. Soft Delete Pattern

### In schema: every tenant table has `deletedAt` (nullable timestamp)

### Drizzle query wrapper
```typescript
import { isNull } from 'drizzle-orm';

// Always add this to queries — never query soft-deleted records
const notDeleted = isNull(tasks.deletedAt);

// List tasks (automatically excludes deleted)
const taskList = await db
  .select()
  .from(tasks)
  .where(and(
    eq(tasks.orgId, tenant.orgId),
    eq(tasks.projectId, projectId),
    notDeleted,              // <- always include this
  ));

// Soft-delete implementation
const [deleted] = await db
  .update(tasks)
  .set({ deletedAt: new Date(), updatedAt: new Date() })
  .where(and(
    eq(tasks.id, taskId),
    eq(tasks.orgId, tenant.orgId),
    isNull(tasks.deletedAt),  // prevent double-delete
  ))
  .returning({ id: tasks.id });

if (!deleted) throw new NotFoundError('Task not found');
```

## 6. Optimistic Locking

```typescript
// Client must send { version: N } in update body
// DB increments version on each update

const [updated] = await db
  .update(tasks)
  .set({
    title:     body.title,
    status:    body.status,
    version:   sql`${tasks.version} + 1`,  // increment
    updatedAt: new Date(),
  })
  .where(and(
    eq(tasks.id, taskId),
    eq(tasks.orgId, tenant.orgId),
    eq(tasks.version, body.version),  // optimistic lock check
    isNull(tasks.deletedAt),
  ))
  .returning();

if (!updated) {
  // Either not found OR version mismatch
  // Check which:
  const exists = await db.select({ id: tasks.id })
    .from(tasks)
    .where(and(eq(tasks.id, taskId), eq(tasks.orgId, tenant.orgId)))
    .limit(1);

  if (!exists.length) throw new NotFoundError('Task not found');
  throw new ConflictError('Task was modified by another request. Refresh and retry.');
}
```

## 7. Cursor-Based Pagination

### How it works
- Sort by `(createdAt DESC, id DESC)` — two-column sort for stability
- Cursor = base64-encoded JSON `{ createdAt: string, id: string }`
- Next page: `WHERE (createdAt, id) < (cursor.createdAt, cursor.id)` — row comparison

```typescript
// src/lib/pagination.ts
import { and, lt, or, eq, desc } from 'drizzle-orm';
import { AnyColumn } from 'drizzle-orm';

export interface CursorPayload {
  createdAt: string;
  id: string;
}

export function encodeCursor(payload: CursorPayload): string {
  return Buffer.from(JSON.stringify(payload)).toString('base64url');
}

export function decodeCursor(cursor: string): CursorPayload {
  return JSON.parse(Buffer.from(cursor, 'base64url').toString('utf8'));
}

// Build Drizzle WHERE clause for keyset pagination
// Equivalent SQL: WHERE (created_at, id) < (cursor.createdAt, cursor.id)
export function cursorWhere(
  createdAtCol: AnyColumn,
  idCol: AnyColumn,
  cursor: CursorPayload
) {
  return or(
    lt(createdAtCol, new Date(cursor.createdAt)),
    and(
      eq(createdAtCol, new Date(cursor.createdAt)),
      lt(idCol, cursor.id),
    )
  );
}
```

## 8. Full-Text Search

```typescript
// src/lib/search.ts
import { sql } from 'drizzle-orm';

// Uses pg_trgm similarity — works with the GIN index created in migration
export function trgmSearch(column: string, query: string) {
  // We use sql`` here because Drizzle doesn't have a built-in trgm operator.
  // The % operator uses the GIN trigram index; ILIKE would cause a seq scan.
  return sql`${sql.identifier(column)} % ${query}`;
}
```

## 9. Audit Log Middleware

```typescript
// src/middleware/auditLog.ts
// Captures the "after" state from the response body and writes to audit_events

export function withAudit(action: 'created' | 'updated' | 'deleted', resourceType: string) {
  return (req: Request, res: Response, next: NextFunction) => {
    // Capture original json method
    const originalJson = res.json.bind(res);

    res.json = function(body: unknown) {
      // Only audit successful mutations
      if (res.statusCode >= 200 && res.statusCode < 300 && body) {
        const resourceId = (body as any).id ?? req.params.id;
        if (resourceId && req.tenant) {
          // Fire-and-forget audit write (don't block the response)
          db.insert(auditEvents).values({
            orgId:        req.tenant.orgId,
            userId:       req.tenant.userId,
            action,
            resourceType,
            resourceId,
            before:       (req as any).__auditBefore ?? null,
            after:        action === 'deleted' ? null : body,
          }).catch(err => console.error('Audit log write failed:', err));
        }
      }
      return originalJson(body);
    };

    next();
  };
}
```
